import { getIssuesData } from '../../../utils/issues';
import CategoryList from '../../../components/CategoryList.astro';

export async function getStaticProps() {
  const { issues, categories } = await getIssuesData();
  return { props: { issues, categories } };
}

このページにはすべてのWikiコンテンツが一覧表示されています。記事の内容に興味がある場合はタイトルをクリックして詳細を確認してください。

<CategoryList 
  issues={frontmatter.getStaticProps.issues} 
  title="すべてのページ" 
  showViewAll={false}
/>
